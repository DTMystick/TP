Ex 11/ select count(numLivre) as nbrLivre, anneeImpression from livre group by anneeImpression; 2/ select min(nombrePages) as minPage, max(nombrePages) as maxPage, anneeImpression     from livre group by anneeImpression;3/ select avg(nombrePages) as moyennePage, anneeImpression from livre group by anneeImpression     having anneeImpression >= 2000; 4/ select count(numL) as livreEcrit, numAut from ecrit group by numAut; 5/ select count(nom) as editeur, ville from editeur group by ville; 6/ select count(numL) as livreEcrit, numE from edite group by numE; Ex 21/ select count(numLivre) as livreEcrit, anneeImpression from auteur, ecrit where auteur.numAuteur = ecrit.numAut and numAut = 3 group by anneeImpression; 2/ select count(numLivre) as livreEcrit, anneeImpression from livre, auteur, ecrit where livre.numLivre = ecrit.numL     and auteur.numAuteur = ecrit.numAut and nom like 'Himes' group by anneeImpression; 3/ select count(numLivre) as livreEcrit, anneeImpression from livre, auteur, ecrit where livre.numLivre = ecrit.numL     and auteur.numAuteur = ecrit.numAut and nom like 'Himes' group by anneeImpression having anneeImpression >= 2000; 4/ select count(numLivre) as livreEdite, ville from editeur, edite where editeur.numE = edite.numE group by ville; Ex 31/  select numLivre from livre where nombrePages > 300 and exists (select numLivre from livre      where anneeImpression < 2003); 2/ select numLivre from livre l1 where nombrePages > 300 and exists (select numLivre from livre l2     where anneeImpression < 2003 and l1.numLivre = l2.numLivre); 3/ select numLivre from livre l1 where nombrePages > 300 and not exists (select numLivre from livre l2     where anneeImpression < 2003 and l1.numLivre = l2.numLivre);
